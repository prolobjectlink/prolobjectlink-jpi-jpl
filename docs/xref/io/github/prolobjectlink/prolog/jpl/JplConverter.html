69  	public final PrologTerm toTerm(Term prologTerm) {
70  		if (prologTerm.isAtom()) {
71  			String functor = prologTerm.name();
72  			if (functor.equals("nil")) {
73  				return new JplNil(provider);
74  			} else if (functor.equals("!")) {
75  				return new JplCut(createProvider());
76  			} else if (functor.equals("fail")) {
77  				return new JplFail(provider);
78  			} else if (functor.equals("true")) {
79  				return new JplTrue(provider);
80  			} else if (functor.equals("false")) {
81  				return new JplFalse(provider);
82  			} else if (functor.equals("[]")) {
83  				return new JplEmpty(provider);
84  			}
85  			return new JplAtom(provider, functor);
86  		} else if (prologTerm.equals(JplList.EMPTY)) {
87  			return new JplEmpty(provider);
88  		} else if (prologTerm.isFloat()) {
89  			return new JplFloat(provider, ((Float) prologTerm).floatValue());
90  		} else if (prologTerm.isInteger()) {
91  			try {
92  				return new JplInteger(provider, ((Integer) prologTerm).intValue());
93  			} catch (JPLException e) {
94  				return new JplLong(provider, ((Integer) prologTerm).longValue());
95  			}
96  		} else if (prologTerm.isVariable()) {
97  			String name = ((Variable) prologTerm).name();
98  			PrologVariable variable = sharedVariables.get(name);
99  			if (variable == null) {
100 				variable = new JplVariable(provider, name);
101 				sharedVariables.put(variable.getName(), variable);
102 			}
103 			return variable;
104 		} else if (prologTerm.hasFunctor(".", 2)) {
105 			Term[] a = new Term[0];
106 			List<Term> l = new ArrayList<Term>();
107 			Term ptr = prologTerm;
108 			while (!ptr.isVariable() && ptr.hasFunctor(".", 2)) {
109 				l.add(ptr.arg(1));
110 				ptr = ptr.arg(2);
111 			}
112 			return new JplList(provider, l.toArray(a));
113 		} else if (prologTerm.isCompound()) {
114 
115 			Compound compound = (Compound) prologTerm;
116 			int arity = compound.arity();
117 			String functor = compound.name();
118 			Term[] arguments = new Term[arity];
119 
120 			// object reference
121 			if (functor.equals("@") && arity == 1) {
122 				return new JplReference(provider, compound);
123 			}
124 
125 			// expressions
126 			else if (arity == 2) {
127 				String key = "LIST";
128 				String opQuery = "findall(OP,current_op(_,_,OP)," + key + ")";
129 				Query query = new Query(opQuery);
130 				if (query.hasSolution()) {
131 
132 					Term term = (Term) query.oneSolution().get(key);
133 					Term[] termArray = term.toTermArray();
134 					for (Term termArray1 : termArray) {
135 						if (termArray1.name().equals(functor)) {
136 							Term left = compound.arg(1);
137 							Term right = compound.arg(2);
138 							return new JplStructure(provider, left, functor, right);
139 						}
140 					}
141 
142 				}
143 				query.close();
144 			}
145 
146 			for (int i = 0; i < arity; i++) {
147 				arguments[i] = compound.arg(i + 1);
148 			}
149 			return new JplStructure(provider, functor, arguments);
150 
151 		}
152 
153 		throw new UnknownTermError(prologTerm);
154 	}

156 	public final Term fromTerm(PrologTerm term) {
157 		switch (term.getType()) {
158 		case NIL_TYPE:
159 			return new Atom("nil");
160 		case CUT_TYPE:
161 			return new Atom("!");
162 		case FAIL_TYPE:
163 			return new Atom("fail");
164 		case TRUE_TYPE:
165 			return new Atom("true");
166 		case FALSE_TYPE:
167 			return new Atom("false");
168 		case ATOM_TYPE:
169 			return new Atom(removeQuoted(((PrologAtom) term).getStringValue()));
170 		case FLOAT_TYPE:
171 			return new Float(((PrologFloat) term).getFloatValue());
172 		case INTEGER_TYPE:
173 			return new Integer(((PrologInteger) term).getIntegerValue());
174 		case DOUBLE_TYPE:
175 			return new Float(((PrologDouble) term).getDoubleValue());
176 		case LONG_TYPE:
177 			return new Integer(((PrologLong) term).getLongValue());
178 		case VARIABLE_TYPE:
179 			String name = ((PrologVariable) term).getName();
180 			Term variable = sharedPrologVariables.get(name);
181 			if (variable == null) {
182 				variable = new Variable(name);
183 				sharedPrologVariables.put(name, variable);
184 			}
185 			return variable;
186 		case LIST_TYPE:
187 			PrologTerm[] array = term.getArguments();
188 			Term list = JplEmpty.EMPTY;
189 			for (int i = array.length - 1; i >= 0; --i) {
190 				list = new Compound(".", new Term[] { fromTerm(array[i]), list });
191 			}
192 			return list;
193 		case STRUCTURE_TYPE:
194 			String functor = term.getFunctor();
195 			Term[] arguments = fromTermArray(((PrologStructure) term).getArguments());
196 			return new Compound(functor, arguments);
197 		case OBJECT_TYPE:
198 			return JPL.newJRef(term.getObject());
199 		default:
200 			throw new UnknownTermError(term);
201 		}
202 	}

204 	public final Term[] fromTermArray(PrologTerm[] terms) {
205 		Term[] prologTerms = new Term[terms.length];
206 		for (int i = 0; i < terms.length; i++) {
207 			prologTerms[i] = fromTerm(terms[i]);
208 		}
209 		return prologTerms;
210 	}

212 	public final Term fromTerm(PrologTerm head, PrologTerm[] body) {
213 		Term clauseHead = fromTerm(head);
214 		if (body != null && body.length > 0) {
215 			Term clauseBody = fromTerm(body[body.length - 1]);
216 			for (int i = body.length - 2; i >= 0; --i) {
217 				clauseBody = new Compound(",", new Term[] { fromTerm(body[i]), clauseBody });
218 			}
219 			return new Compound(":-", new Term[] { clauseHead, clauseBody });
220 		}
221 		return clauseHead;
222 	}
